120  public:
121  std::shared_ptr<nmodl::ast::Program> parse_stream(py::object object) {
122  py::object tiob = py::module::import("io").attr("TextIOBase");
123  if (py::isinstance(object, tiob)) {
124  py::detail::pythonibuf<py::str> buf(object);
125  std::istream istr(&buf);
126  return NmodlDriver::parse_stream(istr);
127  } else {
128  py::detail::pythonibuf<py::bytes> buf(object);
129  std::istream istr(&buf);
130  return NmodlDriver::parse_stream(istr);
131  }
132  }
