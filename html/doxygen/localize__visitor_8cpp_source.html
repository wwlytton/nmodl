19 bool LocalizeVisitor::node_for_def_use_analysis(ast::Node* node) {
20  auto type = node->get_node_type();
21 
22  /**
23  * Blocks where we should compute def-use chains. We are excluding
24  * procedures and functions because we expect those to be "inlined".
25  * If procedure/function is not inlined then DefUse pass returns
26  * result as "Use". So it's safe.
27  */
28  // clang-format off
29  const std::vector<ast::AstNodeType> blocks_to_analyze = {
30  ast::AstNodeType::INITIAL_BLOCK,
31  ast::AstNodeType::BREAKPOINT_BLOCK,
32  ast::AstNodeType::CONSTRUCTOR_BLOCK,
33  ast::AstNodeType::DESTRUCTOR_BLOCK,
34  ast::AstNodeType::DERIVATIVE_BLOCK,
35  ast::AstNodeType::LINEAR_BLOCK,
36  ast::AstNodeType::NON_LINEAR_BLOCK,
37  ast::AstNodeType::DISCRETE_BLOCK,
38  ast::AstNodeType::PARTIAL_BLOCK,
39  ast::AstNodeType::NET_RECEIVE_BLOCK,
40  ast::AstNodeType::TERMINAL_BLOCK,
41  ast::AstNodeType::BA_BLOCK,
42  ast::AstNodeType::FOR_NETCON,
43  ast::AstNodeType::BEFORE_BLOCK,
44  ast::AstNodeType::AFTER_BLOCK,
45  };
46  // clang-format on
47  auto it = std::find(blocks_to_analyze.begin(), blocks_to_analyze.end(), type);
48  auto node_to_analyze = !(it == blocks_to_analyze.end());
49  auto solve_procedure = is_solve_procedure(node);
50  return (node_to_analyze || solve_procedure);
51 }

58  if (node->is_procedure_block()) {
59  auto symbol = program_symtab->lookup(node->get_node_name());
60  if (symbol && symbol->has_any_property(NmodlType::to_solve)) {
61  return true;
62  }
63  }
64  return false;

67 std::vector<std::string> LocalizeVisitor::variables_to_optimize() {
68  // clang-format off
69  const NmodlType excluded_var_properties = NmodlType::extern_var
70  | NmodlType::extern_neuron_variable
71  | NmodlType::read_ion_var
72  | NmodlType::write_ion_var
73  | NmodlType::prime_name

75  | NmodlType::pointer_var
76  | NmodlType::bbcore_pointer_var

80  NmodlType global_var_properties = NmodlType::range_var
81  | NmodlType::assigned_definition
82  | NmodlType::param_assign;
83  // clang-format on
84 
85 
86  /**
87  * \todo Voltage v can be global variable as well as external. In order
88  * to avoid optimizations, we need to handle this case properly
89  * \todo Instead of ast node, use symbol properties to check variable type
90  */
91  auto variables = program_symtab->get_variables_with_properties(global_var_properties);
92 
93  std::vector<std::string> result;
94  for (auto& variable: variables) {
95  if (!variable->has_any_property(excluded_var_properties)) {
96  result.push_back(variable->get_name());
97  }
98  }
99  return result;

102 void LocalizeVisitor::visit_program(ast::Program* node) {
103  /// symtab visitor pass need to be run before
104  program_symtab = node->get_symbol_table();
105  if (program_symtab == nullptr) {
106  logger->warn("LocalizeVisitor :: symbol table is not setup, returning");
107  return;
108  }
109 
110  auto variables = variables_to_optimize();
111  for (const auto& varname: variables) {
112  const auto& blocks = node->get_blocks();
113  std::map<DUState, std::vector<std::shared_ptr<ast::Node>>> block_usage;
114 
115  /// compute def use chains
116  for (auto block: blocks) {
117  if (node_for_def_use_analysis(block.get())) {
118  DefUseAnalyzeVisitor v(program_symtab, ignore_verbatim);
119  auto usages = v.analyze(block.get(), varname);
120  auto result = usages.eval();
121  block_usage[result].push_back(block);
122  }
123  }
124 
125  /// as we are doing global analysis, if any global block is "using"
126  /// variable then we can't localize the variable
127  auto it = block_usage.find(DUState::U);
128  if (it == block_usage.end()) {
129  logger->debug("LocalizeVisitor : localized variable {}", varname);
130 
131  /// all blocks that are have either definition or conditional definition
132  /// need local variable
133  for (auto state: {DUState::D, DUState::CD}) {
134  for (auto& block: block_usage[state]) {
135  auto block_ptr = dynamic_cast<ast::Block*>(block.get());
136  auto statement_block = block_ptr->get_statement_block();
137  ast::LocalVar* variable;
138  auto symbol = program_symtab->lookup(varname);
139 
140  if (symbol->is_array()) {
141  variable = add_local_variable(statement_block.get(),
142  varname,
143  symbol->get_length());
144  } else {
145  variable = add_local_variable(statement_block.get(), varname);
146  }
147 
148  /// mark variable as localized in global symbol table
149  symbol->mark_localized();
150 
151  /// insert new symbol in the symbol table of current block
152  auto symtab = statement_block->get_symbol_table();
153  auto new_symbol = std::make_shared<Symbol>(varname, variable);
154  new_symbol->add_property(NmodlType::local_var);
155  new_symbol->mark_created();
156  symtab->insert(new_symbol);
157  }
158  }
159  }
160  }
161 }
