30 /**
31  * How symbols are stored in NEURON? See notes written in markdown file.
32  *
33  * Some variables get printed by iterating over symbol table in mod2c.
34  * The example of this is thread variables (and also ions?). In this
35  * case we must have to arrange order if we are going keep compatibility
36  * with NEURON.
37  *
38  * Suppose there are three global variables: bcd, abc, abd, abe
39  * They will be in the 'a' bucket in order:
40  * abe, abd, abc
41  * and in 'b' bucket
42  * bcd
43  * So when we print thread variables, we first have to sort in the opposite
44  * order in which they come and then again order by first character in increasing
45  * order.
46  *
47  * Note that variables in double array do not need this transformation
48  * and it seems like they should just follow definition order.
49  */
50 void CodegenHelperVisitor::sort_with_mod2c_symbol_order(std::vector<SymbolType>& symbols) {
51  /// first sort by global id to get in reverse order
52  std::sort(symbols.begin(),
53  symbols.end(),
54  [](const SymbolType& first, const SymbolType& second) -> bool {
55  return first->get_id() > second->get_id();
56  });
57 
58  /// now order by name (to be same as neuron's bucket)
59  std::sort(symbols.begin(),
60  symbols.end(),
61  [](const SymbolType& first, const SymbolType& second) -> bool {
62  return first->get_name()[0] < second->get_name()[0];
63  });
64 }

67 /**
68  * Find all ions used in mod file
69  */
70 void CodegenHelperVisitor::find_ion_variables() {
71  /// name of the ions used
72  auto ion_vars = psymtab->get_variables_with_properties(NmodlType::useion);
73  /// read variables from all ions
74  auto read_ion_vars = psymtab->get_variables_with_properties(NmodlType::read_ion_var);
75  /// write variables from all ions
76  auto write_ion_vars = psymtab->get_variables_with_properties(NmodlType::write_ion_var);
77 
78  /**
79  * Check if given variable belongs to given ion.
80  * For example, eca belongs to ca ion, nai belongs to na ion.
81  * We just check if we exclude first/last char, if that is ion name.
82  */
83  auto ion_variable = [](std::string var, std::string ion) -> bool {
84  auto len = var.size() - 1;
85  return (var.substr(1, len) == ion || var.substr(0, len) == ion);
86  };
87 
88  /// iterate over all ion types and construct the Ion objects
89  for (auto& ion_var: ion_vars) {
90  auto ion_name = ion_var->get_name();
91  Ion ion(ion_name);
92  for (auto& read_var: read_ion_vars) {
93  auto var = read_var->get_name();
94  if (ion_variable(var, ion_name)) {
95  ion.reads.push_back(var);
96  }
97  }
98  for (auto& write_var: write_ion_vars) {
99  auto varname = write_var->get_name();
100  if (ion_variable(varname, ion_name)) {
101  ion.writes.push_back(varname);
102  if (ion.is_intra_cell_conc(varname) || ion.is_extra_cell_conc(varname)) {
103  ion.need_style = true;
104  info.write_concentration = true;
105  }
106  }
107  }
108  info.ions.push_back(std::move(ion));
109  }
110 
111  /// once ions are populated, we can find all currents
112  auto vars = psymtab->get_variables_with_properties(NmodlType::nonspecific_cur_var);
113  for (auto& var: vars) {
114  info.currents.push_back(var->get_name());
115  }
116  vars = psymtab->get_variables_with_properties(NmodlType::electrode_cur_var);
117  for (auto& var: vars) {
118  info.currents.push_back(var->get_name());
119  }
120  for (auto& ion: info.ions) {
121  for (auto& var: ion.writes) {
122  if (ion.is_ionic_current(var)) {
123  info.currents.push_back(var);
124  }
125  }
126  }
127 
128  /// check if worte_conc(...) will be needed
129  for (const auto& ion: info.ions) {
130  for (const auto& var: ion.writes) {
131  if (!ion.is_ionic_current(var) && !ion.is_rev_potential(var)) {
132  info.require_wrote_conc = true;
133  }
134  }
135  }
136 }

139 /**
140  * Find non-range variables i.e. ones that are not belong to per instance allocation
141  *
142  * Certain variables like pointers, global, parameters are not necessary to be per
143  * instance variables. NEURON apply certain rules to determine which variables become
144  * thread, static or global variables. Here we construct those variables.
145  */
146 void CodegenHelperVisitor::find_non_range_variables() {
147  /**
148  * Top local variables are local variables appear in global scope. All local
149  * variables in program symbol table are in global scope.
150  */
151  info.constant_variables = psymtab->get_variables_with_properties(NmodlType::constant_var);
152  info.top_local_variables = psymtab->get_variables_with_properties(NmodlType::local_var);
153 
154  /**
155  * All global variables remain global if mod file is not marked thread safe.
156  * Otherwise, global variables written at least once gets promoted to thread variables.
157  */
158 
159  std::string variables;
160 
161  auto vars = psymtab->get_variables_with_properties(NmodlType::global_var);
162  for (auto& var: vars) {
163  if (info.thread_safe && var->get_write_count() > 0) {
164  var->mark_thread_safe();
165  info.thread_variables.push_back(var);
166  info.thread_var_data_size += var->get_length();
167  variables += " " + var->get_name();
168  } else {
169  info.global_variables.push_back(var);
170  }
171  }
172 
173  /**
174  * If parameter is not a range and used only as read variable then it becomes global
175  * variable. To qualify it as thread variable it must be be written at least once and
176  * mod file must be marked as thread safe.
177  * To exclusively get parameters only, we exclude all other variables (in without)
178  * and then sort them with neuron/mod2c order.
179  */
180  // clang-format off
181  auto with = NmodlType::param_assign;
182  auto without = NmodlType::range_var
183  | NmodlType::assigned_definition
184  | NmodlType::global_var
185  | NmodlType::pointer_var
186  | NmodlType::bbcore_pointer_var
187  | NmodlType::read_ion_var
188  | NmodlType::write_ion_var;
189  // clang-format on
190  vars = psymtab->get_variables(with, without);
191  for (auto& var: vars) {
192  // some variables like area and diam are declared in parameter
193  // block but they are not global
194  if (var->get_name() == naming::DIAM_VARIABLE || var->get_name() == naming::AREA_VARIABLE ||
195  var->has_any_property(NmodlType::extern_neuron_variable)) {
196  continue;
197  }
198 
199  // if model is thread safe and if parameter is being written then
200  // those variables should be promoted to thread safe variable
201  if (info.thread_safe && var->get_write_count() > 0) {
202  var->mark_thread_safe();
203  info.thread_variables.push_back(var);
204  info.thread_var_data_size += var->get_length();
205  } else {
206  info.global_variables.push_back(var);
207  }
208  }
209  sort_with_mod2c_symbol_order(info.thread_variables);
210 
211  /**
212  * \todo Below we calculate thread related id and sizes. This will
213  * need to do from global analysis pass as here we are handling
214  * top local variables, global variables, derivimplicit method.
215  * There might be more use cases with other solver methods.
216  */
217 
218  /**
219  * If derivimplicit is used, then first three thread ids get assigned to:
220  * 1st thread is used for: deriv_advance
221  * 2nd thread is used for: dith
222  * 3rd thread is used for: newtonspace
223  *
224  * slist and dlist represent the offsets for prime variables used. For
225  * euler or derivimplicit methods its always first number.
226  */
227  if (info.derivimplicit_used) {
228  info.derivimplicit_var_thread_id = 0;
229  info.thread_data_index = 3;
230  info.derivimplicit_list_num = 1;
231  info.thread_callback_register = true;
232  }
233 
234  /// next thread id is allocated for top local variables
235  if (info.vectorize && !info.top_local_variables.empty()) {
236  info.top_local_thread_id = info.thread_data_index++;
237  info.thread_callback_register = true;
238  }
239 
240  /// next thread id is allocated for thread promoted variables
241  if (info.vectorize && !info.thread_variables.empty()) {
242  info.thread_var_thread_id = info.thread_data_index++;
243  info.thread_callback_register = true;
244  }
245 
246  /// find total size of local variables in global scope
247  for (auto& var: info.top_local_variables) {
248  info.top_local_thread_size += var->get_length();
249  }
250 
251  /// find number of prime variables and total size
252  auto primes = psymtab->get_variables_with_properties(NmodlType::prime_name);
253  info.num_primes = primes.size();
254  for (auto& variable: primes) {
255  info.primes_size += variable->get_length();
256  }
257 
258  /// find pointer or bbcore pointer variables
259  // clang-format off
260  auto properties = NmodlType::pointer_var
261  | NmodlType::bbcore_pointer_var;
262  // clang-format on
263  info.pointer_variables = psymtab->get_variables_with_properties(properties);
264 
265  // find special variables like diam, area
266  // clang-format off
267  properties = NmodlType::assigned_definition
268  | NmodlType::param_assign;
269  vars = psymtab->get_variables_with_properties(properties);
270  for (auto& var : vars) {
271  if (var->get_name() == naming::AREA_VARIABLE) {
272  info.area_used = true;
273  }
274  if (var->get_name() == naming::DIAM_VARIABLE) {
275  info.diam_used = true;
276  }
277  }
278  // clang-format on
279 }

281 /**
282  * Find range variables i.e. ones that are belong to per instance allocation
283  *
284  * In order to be compatible with NEURON, we need to print range variables in
285  * certain order. For example, range variables which are parameters comes first.
286  * Also, there is difference between declaration order vs. definition order. For
287  * example, POINTER variable in NEURON block is just declaration and doesn't
288  * determine the order in which they will get printed. Below we query symbol table
289  * and order all instance variables into certain order.
290  */
291 void CodegenHelperVisitor::find_range_variables() {
292  /// comparator to decide the order based on definition
293  auto comparator = [](const SymbolType& first, const SymbolType& second) -> bool {
294  return first->get_definition_order() < second->get_definition_order();
295  };
296 
297  /**
298  * First come parameters which are range variables.
299  */
300  // clang-format off
301  auto with = NmodlType::range_var
302  | NmodlType::param_assign;
303  auto without = NmodlType::global_var
304  | NmodlType::pointer_var
305  | NmodlType::bbcore_pointer_var
306  | NmodlType::state_var;
307  // clang-format on
308  info.range_parameter_vars = psymtab->get_variables(with, without);
309  std::sort(info.range_parameter_vars.begin(), info.range_parameter_vars.end(), comparator);
310 
311  /**
312  * Second come assigned variables which are range variables.
313  */
314  // clang-format off
315  with = NmodlType::range_var
316  | NmodlType::assigned_definition;
317  without = NmodlType::global_var
318  | NmodlType::pointer_var
319  | NmodlType::bbcore_pointer_var
320  | NmodlType::state_var
321  | NmodlType::param_assign;
322  // clang-format on
323  info.range_assigned_vars = psymtab->get_variables(with, without);
324  std::sort(info.range_assigned_vars.begin(), info.range_assigned_vars.end(), comparator);
325 
326  /**
327  * Third come state variables. All state variables are kind of range by default.
328  * Note that some mod files like CaDynamics_E2.mod use cai as state variable
329  * and those are not considered as range+state variables while printing instance
330  * variables. Such read/write ion variables are assigned variables and hence they
331  * will be printed at laster stage.
332  * \todo Need to validate with more models and mod2c details.
333  */
334  // clang-format off
335  with = NmodlType::state_var;
336  without = NmodlType::global_var
337  | NmodlType::pointer_var
338  | NmodlType::bbcore_pointer_var
339  | NmodlType::read_ion_var
340  | NmodlType::write_ion_var;
341  // clang-format on
342  info.state_vars = psymtab->get_variables(with, without);
343  std::sort(info.state_vars.begin(), info.state_vars.end(), comparator);
344 
345  /**
346  * Remaining variables are:
347  * - all assigned variables without range
348  * - read ion variables which appear in parameter or assigned block
349  * - state variables which are not range but with ion variable of read/write type
350  */
351 
352  /**
353  * first get assigned definition without read ion variables
354  */
355  // clang-format off
356  with = NmodlType::assigned_definition;
357  without = NmodlType::global_var
358  | NmodlType::pointer_var
359  | NmodlType::bbcore_pointer_var
360  | NmodlType::state_var
361  | NmodlType::range_var
362  | NmodlType::extern_neuron_variable
363  | NmodlType::read_ion_var;
364  // clang-format on
365  info.assigned_vars = psymtab->get_variables(with, without);
366 
367  /**
368  * Now just use read-ion variables because every read-ion variable
369  * must be part of either assigned or parameter block. Otherwise code is not
370  * compiled anyway.
371  */
372  // clang-format off
373  with = NmodlType::read_ion_var;
374  without = NmodlType::global_var
375  | NmodlType::pointer_var
376  | NmodlType::bbcore_pointer_var
377  | NmodlType::state_var
378  | NmodlType::range_var
379  | NmodlType::extern_neuron_variable;
380  // clang-format on
381  auto variables = psymtab->get_variables(with, without);
382  info.assigned_vars.insert(info.assigned_vars.end(), variables.begin(), variables.end());
383 
384  /*
385  * We want to have state variables which are read or write ion variables.
386  * This needs to be separated from other state variables because mod2c
387  * treat them separately for ordering.
388  */
389  // clang-format off
390  with = NmodlType::state_var;
391  without = NmodlType::global_var
392  | NmodlType::pointer_var
393  | NmodlType::bbcore_pointer_var
394  | NmodlType::range_var
395  | NmodlType::extern_neuron_variable;
396  // clang-format on
397  variables = psymtab->get_variables(with, without);
398  for (auto& variable: variables) {
399  // clang-format off
400  auto properties = NmodlType::read_ion_var
401  | NmodlType::write_ion_var;
402  // clang-format on
403  if (variable->has_any_property(properties)) {
404  info.ion_state_vars.push_back(variable);
405  info.assigned_vars.push_back(variable);
406  }
407  }
408 }

532 /**
533  * Visit statement block and find prime symbols appear in derivative block
534  *
535  * Equation statements in derivative block has prime on the lhs. The order
536  * of primes could be different that declaration state block. Also, not all
537  * state variables need to appear in equation block. In this case, we want
538  * to find out the the primes in the order of equation definition. This is
539  * just to keep the same order as neuron implementation.
540  *
541  * The primes are already solved and replaced by Dstate or name. And hence
542  * we need to check if the lhs variable is derived from prime name. If it's
543  * Dstate then we have to lookup state to find out corresponding symbol. This
544  * is because prime_variables_by_order should contain state variable name and
545  * not the one replaced by solver pass.
546  */
547 void CodegenHelperVisitor::visit_statement_block(ast::StatementBlock* node) {
548  auto statements = node->get_statements();
549  for (auto& statement: statements) {
550  statement->accept(*this);
551  if (under_derivative_block && assign_lhs &&
552  (assign_lhs->is_name() || assign_lhs->is_var_name())) {
553  auto name = assign_lhs->get_node_name();
554  auto symbol = psymtab->lookup(name);
555  if (symbol != nullptr) {
556  auto is_prime = symbol->has_any_property(NmodlType::prime_name);
557  auto from_state = symbol->has_any_status(Status::from_state);
558  if (is_prime || from_state) {
559  if (from_state) {
560  symbol = psymtab->lookup(name.substr(1, name.size()));
561  }
562  info.prime_variables_by_order.push_back(symbol);
563  info.num_equations++;
564  }
565  }
566  }
567  assign_lhs = nullptr;
568  }
569 }

612 void CodegenHelperVisitor::visit_program(ast::Program* node) {
613  psymtab = node->get_symbol_table();
614  auto blocks = node->get_blocks();
615  for (auto& block: blocks) {
616  info.top_blocks.push_back(block.get());
617  if (block->is_verbatim()) {
618  info.top_verbatim_blocks.push_back(block.get());
619  }
620  }
621  node->visit_children(*this);
622  find_range_variables();
623  find_non_range_variables();
624  find_ion_variables();
625  find_table_variables();
626 }
