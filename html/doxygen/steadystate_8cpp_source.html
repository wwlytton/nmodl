33 std::vector<std::string> run_steadystate_visitor(
34  const std::string& text,
35  std::vector<AstNodeType> ret_nodetypes = {AstNodeType::SOLVE_BLOCK,
36  AstNodeType::DERIVATIVE_BLOCK}) {
37  std::vector<std::string> results;
38  // construct AST from text
39  NmodlDriver driver;
40  auto ast = driver.parse_string(text);
41 
42  // construct symbol table from AST
43  SymtabVisitor().visit_program(ast.get());
44 
45  // unroll loops and fold constants
46  ConstantFolderVisitor().visit_program(ast.get());
47  LoopUnrollVisitor().visit_program(ast.get());
48  ConstantFolderVisitor().visit_program(ast.get());
49  SymtabVisitor().visit_program(ast.get());
50 
51  // Run kinetic block visitor first, so any kinetic blocks
52  // are converted into derivative blocks
53  KineticBlockVisitor().visit_program(ast.get());
54  SymtabVisitor().visit_program(ast.get());
55 
56  // run SteadystateVisitor on AST
57  SteadystateVisitor().visit_program(ast.get());
58 
59  // run lookup visitor to extract results from AST
60  auto res = AstLookupVisitor().lookup(ast.get(), ret_nodetypes);
61  for (const auto& r: res) {
62  results.push_back(to_nmodl(r.get()));
63  }
64  return results;
65 }
