71 std::string CodegenIspcVisitor::double_to_string(double value) {
72  if (std::ceil(value) == value) {
73  return "{:.1f}d"_format(value);
74  }
75  if ((value <= 1.0) && (value >= -1.0)) {
76  return "{:f}d"_format(value);
77  } else {
78  auto e = std::log10(std::abs(value));
79  if (e < 0.0) {
80  e = std::ceil(-e);
81  auto m = std::pow(10, e);
82  return "{:f}d-{:d}"_format(value * m, static_cast<int>(e));
83  } else {
84  e = std::floor(e);
85  auto m = std::pow(10, e);
86  return "{:f}d{:d}"_format(value / m, static_cast<int>(e));
87  }
88  }
89 }

324 void CodegenIspcVisitor::print_compute_functions() {
325  for (const auto& function: info.functions) {
326  if (!program_symtab->lookup(function->get_node_name())
327  .get()
328  ->has_all_status(Status::inlined)) {
329  print_function(function);
330  }
331  }
332  for (const auto& procedure: info.procedures) {
333  if (!program_symtab->lookup(procedure->get_node_name())
334  .get()
335  ->has_all_status(Status::inlined)) {
336  print_procedure(procedure);
337  }
338  }
339  if (!emit_fallback[BlockType::NetReceive]) {
340  print_net_receive_kernel();
341  print_net_receive_buffering(false);
342  }
343  if (!emit_fallback[BlockType::Initial]) {
344  print_nrn_init(false);
345  }
346  if (!emit_fallback[BlockType::Equation]) {
347  print_nrn_cur();
348  }
349  if (!emit_fallback[BlockType::State]) {
350  print_nrn_state();
351  }

356 void CodegenIspcVisitor::print_mechanism_global_var_structure(bool wrapper) {
357  std::string decorator = "";
358  if (!wrapper) {
359  decorator = "uniform ";
360  }
361 
362  auto float_type = default_float_data_type();
363  printer->add_newline(2);
364  printer->add_line("/** all global variables */");
365  printer->add_line("struct {} {}"_format(global_struct(), "{"));
366  printer->increase_indent();
367 
368  if (!info.ions.empty()) {
369  for (const auto& ion: info.ions) {
370  auto name = "{}_type"_format(ion.name);
371  printer->add_line("{}int {};"_format(decorator, name));
372  codegen_global_variables.push_back(make_symbol(name));
373  }
374  }
375 
376  if (info.point_process) {
377  printer->add_line("{}int point_type;"_format(decorator));
378  codegen_global_variables.push_back(make_symbol("point_type"));
379  }
380 
381  if (!info.state_vars.empty()) {
382  for (const auto& var: info.state_vars) {
383  auto name = var->get_name() + "0";
384  auto symbol = program_symtab->lookup(name);
385  if (symbol == nullptr) {
386  printer->add_line("{}{} {};"_format(decorator, float_type, name));
387  codegen_global_variables.push_back(make_symbol(name));
388  }
389  }
390  }
391 
392  if (!info.vectorize) {
393  printer->add_line("{}{} v;"_format(decorator, float_type));
394  codegen_global_variables.push_back(make_symbol("v"));
395  }
396 
397  auto& top_locals = info.top_local_variables;
398  if (!info.vectorize && !top_locals.empty()) {
399  for (const auto& var: top_locals) {
400  auto name = var->get_name();
401  auto length = var->get_length();
402  if (var->is_array()) {
403  printer->add_line("{}{} {}[{}];"_format(decorator, float_type, name, length));
404  } else {
405  printer->add_line("{}{} {};"_format(decorator, float_type, name));
406  }
407  codegen_global_variables.push_back(var);
408  }
409  }
410 
411  if (!info.thread_variables.empty()) {
412  printer->add_line("{}int thread_data_in_use;"_format(decorator));
413  printer->add_line(
414  "{}{} thread_data[{}];"_format(decorator, float_type, info.thread_var_data_size));
415  codegen_global_variables.push_back(make_symbol("thread_data_in_use"));
416  auto symbol = make_symbol("thread_data");
417  symbol->set_as_array(info.thread_var_data_size);
418  codegen_global_variables.push_back(symbol);
419  }
420 
421  printer->add_line("{}int reset;"_format(decorator));
422  codegen_global_variables.push_back(make_symbol("reset"));
423 
424  printer->add_line("{}int mech_type;"_format(decorator));
425  codegen_global_variables.push_back(make_symbol("mech_type"));
426 
427  auto& globals = info.global_variables;
428  auto& constants = info.constant_variables;
429 
430  if (!globals.empty()) {
431  for (const auto& var: globals) {
432  auto name = var->get_name();
433  auto length = var->get_length();
434  if (var->is_array()) {
435  printer->add_line("{}{} {}[{}];"_format(decorator, float_type, name, length));
436  } else {
437  printer->add_line("{}{} {};"_format(decorator, float_type, name));
438  }
439  codegen_global_variables.push_back(var);
440  }
441  }
442 
443  if (!constants.empty()) {
444  for (const auto& var: constants) {
445  auto name = var->get_name();
446  auto value_ptr = var->get_value();
447  printer->add_line("{}{} {};"_format(decorator, float_type, name));
448  codegen_global_variables.push_back(var);
449  }
450  }
451 
452  if (info.primes_size != 0) {
453  printer->add_line("int* {}slist1;"_format(decorator));
454  printer->add_line("int* {}dlist1;"_format(decorator));
455  codegen_global_variables.push_back(make_symbol("slist1"));
456  codegen_global_variables.push_back(make_symbol("dlist1"));
457  if (info.derivimplicit_used) {
458  printer->add_line("int* {}slist2;"_format(decorator));
459  codegen_global_variables.push_back(make_symbol("slist2"));
460  }
461  }
462 
463  if (info.table_count > 0) {
464  printer->add_line("{}double usetable;"_format(decorator));
465  codegen_global_variables.push_back(make_symbol(naming::USE_TABLE_VARIABLE));
466 
467  for (const auto& block: info.functions_with_table) {
468  auto name = block->get_node_name();
469  printer->add_line("{}{} tmin_{};"_format(decorator, float_type, name));
470  printer->add_line("{}{} mfac_{};"_format(decorator, float_type, name));
471  codegen_global_variables.push_back(make_symbol("tmin_" + name));
472  codegen_global_variables.push_back(make_symbol("mfac_" + name));
473  }
474 
475  for (const auto& variable: info.table_statement_variables) {
476  auto name = "t_" + variable->get_name();
477  printer->add_line("{}* {}{};"_format(float_type, decorator, name));
478  codegen_global_variables.push_back(make_symbol(name));
479  }
480  }
481 
482  if (info.vectorize) {
483  printer->add_line("ThreadDatum* {}ext_call_thread;"_format(decorator));
484  codegen_global_variables.push_back(make_symbol("ext_call_thread"));
485  }
486 
487  printer->decrease_indent();
488  printer->add_line("};");
489 
490  printer->add_newline(1);
491  if (wrapper) {
492  printer->add_line("/** holds object of global variable */");
493  printer->start_block("extern \"C\"");
494  printer->add_line("{} {}_global;"_format(global_struct(), info.mod_suffix));
495  printer->end_block(2);
496  } else {
497  printer->add_line("/** holds object of global variable */");
498  printer->add_line("extern {} {}_global;"_format(global_struct(), info.mod_suffix));
499  }
500 }

636 void CodegenIspcVisitor::determine_target() {

638 
639  if (info.initial_node) {
640  emit_fallback[BlockType::Initial] =
641  !node_lv.lookup(info.initial_node).empty() ||
642  visitor::calls_function(info.initial_node, "net_send") || info.require_wrote_conc;
643  } else {

646  }

650  visitor::calls_function(info.net_receive_node,
651  "net_send");
652  }

660  }
661

664  emit_fallback[BlockType::State] = !node_lv.lookup(info.nrn_state_block).empty();

671 void CodegenIspcVisitor::move_procs_to_wrapper() {
672  auto nameset = std::set<std::string>();
673 
674  auto populate_nameset = [&nameset](ast::Block* block) {
675  auto name_lv = AstLookupVisitor(ast::AstNodeType::NAME);
676  if (block) {
677  auto names = name_lv.lookup(block);
678  for (const auto& name: names) {
679  nameset.insert(name->get_node_name());
680  }
681  }
682  };
683  populate_nameset(info.initial_node);
684  populate_nameset(info.nrn_state_block);
685  populate_nameset(info.breakpoint_node);
686 
687  auto node_lv = AstLookupVisitor(incompatible_node_types);
688  auto target_procedures = std::vector<ast::ProcedureBlock*>();
689  for (auto it = info.procedures.begin(); it != info.procedures.end(); it++) {
690  auto procname = (*it)->get_name()->get_node_name();
691  if (nameset.find(procname) == nameset.end() || !node_lv.lookup(*it).empty()) {
692  wrapper_procedures.push_back(*it);
693  } else {
694  target_procedures.push_back(*it);
695  }
696  }
697  info.procedures = target_procedures;
698  auto target_functions = std::vector<ast::FunctionBlock*>();
699  for (auto it = info.functions.begin(); it != info.functions.end(); it++) {
700  auto procname = (*it)->get_name()->get_node_name();
701  if (nameset.find(procname) == nameset.end() || !node_lv.lookup(*it).empty()) {
702  wrapper_functions.push_back(*it);
703  } else {
704  target_functions.push_back(*it);
705  }
706  }
707  info.functions = target_functions;
708 }
