18 /**
19  * \class IndexRemover
20  * \brief Helper visitor to replace index of array variable with integer
21  *
22  * When loop is unrolled, the index variable like `i` :
23  *
24  * ca[i] <-> ca[i+1]
25  *
26  * has type `Name` in the AST. This needs to be replaced with `Integer`
27  * for optimizations like constant folding. This pass look at name and
28  * binary expressions under index variables.
29  */
30 class IndexRemover: public AstVisitor {
31  private:
32  /// index variable name
33  std::string index;
34 
35  /// integer value of index variable
36  int value;
37 
38  /// true if we are visiting index variable
39  bool under_indexed_name = false;
40 
41  public:
42  IndexRemover(std::string index, int value)
43  : index(index)
44  , value(value) {}
45 
46  /// if expression we are visiting is `Name` then return new `Integer` node
47  std::shared_ptr<ast::Expression> replace_for_name(const std::shared_ptr<ast::Expression> node) {
48  if (node->is_name()) {
49  auto name = std::dynamic_pointer_cast<ast::Name>(node);
50  if (name->get_node_name() == index) {
51  return std::make_shared<ast::Integer>(value, nullptr);
52  }
53  }
54  return node;
55  }
56 
57  virtual void visit_binary_expression(ast::BinaryExpression* node) override {
58  node->visit_children(*this);
59  if (under_indexed_name) {
60  /// first recursively replaces childrens
61  /// replace lhs & rhs if they have matching index variable
62  auto lhs = replace_for_name(node->get_lhs());
63  auto rhs = replace_for_name(node->get_rhs());
64  node->set_lhs(std::move(lhs));
65  node->set_rhs(std::move(rhs));
66  }
67  }
68 
69  virtual void visit_indexed_name(ast::IndexedName* node) override {
70  under_indexed_name = true;
71  node->visit_children(*this);
72  /// once all children are replaced, do the same for index
73  auto length = replace_for_name(node->get_length());
74  node->set_length(std::move(length));
75  under_indexed_name = false;
76  }
77 };

90 /**
91  * Unroll given for loop
92  *
93  * @param node From loop node in the AST
94  * @return expression statement represeing unrolled loop if successfull otherwise nullptr
95  */
96 static std::shared_ptr<ast::ExpressionStatement> unroll_for_loop(
97  const std::shared_ptr<ast::FromStatement> node) {
98  /// loop can be in the form of `FROM i=(0) TO (10)`
99  /// so first unwrap all elements of the loop
100  const auto from = unwrap(node->get_from());
101  const auto to = unwrap(node->get_to());
102  const auto increment = unwrap(node->get_increment());
103 
104  /// we can unroll if iteration space of the loop is known
105  /// after constant folding start, end and increment must be known
106  if (!from->is_integer() || !to->is_integer() ||
107  (increment != nullptr && !increment->is_integer())) {
108  return nullptr;
109  }
110 
111  int start = std::dynamic_pointer_cast<ast::Integer>(from)->eval();
112  int end = std::dynamic_pointer_cast<ast::Integer>(to)->eval();
113  int step = 1;
114  if (increment != nullptr) {
115  step = std::dynamic_pointer_cast<ast::Integer>(increment)->eval();
116  }
117 
118  ast::StatementVector statements;
119  std::string index_var = node->get_node_name();
120  for (int i = start; i <= end; i += step) {
121  /// duplicate loop body and copy all statements to new vector
122  auto new_block = node->get_statement_block()->clone();
123  IndexRemover(index_var, i).visit_statement_block(new_block);
124  statements.insert(statements.end(),
125  new_block->statements.begin(),
126  new_block->statements.end());
127  delete new_block;
128  }
129 
130  /// create new statement representing unrolled loop
131  auto block = new ast::StatementBlock(std::move(statements));
132  return std::make_shared<ast::ExpressionStatement>(block);
133 }

136 /**
137  * Parse verbatim blocks and rename variable if it is used.
138  */
139 void LoopUnrollVisitor::visit_statement_block(ast::StatementBlock* node) {
140  node->visit_children(*this);
141 
142  for (auto iter = node->statements.begin(); iter != node->statements.end(); iter++) {
143  if ((*iter)->is_from_statement()) {
144  auto statement = std::dynamic_pointer_cast<ast::FromStatement>((*iter));
145 
146  /// check if any verbatim block exists
147  auto verbatim_blocks = AstLookupVisitor().lookup(statement.get(),
148  ast::AstNodeType::VERBATIM);
149  if (!verbatim_blocks.empty()) {
150  logger->debug("LoopUnrollVisitor : can not unroll because of verbatim block");
151  continue;
152  }
153 
154  /// unroll loop, replace current statement on successfull unroll
155  auto new_statement = unroll_for_loop(statement);
156  if (new_statement != nullptr) {
157  (*iter) = new_statement;
158  std::string before = to_nmodl(statement.get());
159  std::string after = to_nmodl(new_statement.get());
160  logger->debug("LoopUnrollVisitor : \n {} \n unrolled to \n {}", before, after);
161  }
162  }
163  }
164 }
