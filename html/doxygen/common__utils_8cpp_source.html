27 bool make_path(const std::string& path) {
28  mode_t mode = 0755;
29  int ret = mkdir(path.c_str(), mode);
30  if (ret == 0) {
31  return true;
32  }
33 
34  switch (errno) {
35  case ENOENT:
36  // parent didn't exist, try to create it
37  {
38  int pos = path.find_last_of('/');
39  if (pos == std::string::npos) {
40  return false;
41  }
42  if (!make_path(path.substr(0, pos))) {
43  return false;
44  }
45  }
46  // now, try to create again
47  return 0 == mkdir(path.c_str(), mode);
48 
49  case EEXIST:
50  // done!
51  return is_dir_exist(path);
52 
53  default:
54  auto msg = "Can not create directory " + path;
55  throw std::runtime_error(msg);
56  }
57 }
