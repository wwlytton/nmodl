26 void SympySolverVisitor::init_block_data(ast::Node* node) {

28  expression_statements.clear();
29  eq_system.clear();
30  state_vars_in_block.clear();
31  last_expression_statement = nullptr;
32  block_with_expression_statements = nullptr;
33  eq_system_is_valid = true;
34  conserve_equation.clear();

37  if (auto symtab = node->get_statement_block()->get_symbol_table()) {

49  function_calls.insert(call->get_node_name());

84 ast::StatementVector::iterator SympySolverVisitor::get_solution_location_iterator(
85  ast::StatementVector& statements) {
86  // find out where to insert solutions in statement block
87  // returns iterator pointing to the first element after the last (non)linear eq
88  // so if there are no such elements, it returns statements.end()
89  auto it = statements.begin();
90  if (last_expression_statement != nullptr) {
91  while ((it != statements.end()) &&
92  (std::dynamic_pointer_cast<ast::ExpressionStatement>(*it).get() !=
93  last_expression_statement)) {
94  logger->debug("SympySolverVisitor :: {} != {}",
95  to_nmodl((*it).get()),
96  to_nmodl(last_expression_statement));
97  ++it;
98  }
99  if (it != statements.end()) {
100  logger->debug("SympySolverVisitor :: {} == {}",
101  to_nmodl(std::dynamic_pointer_cast<ast::ExpressionStatement>(*it).get()),
102  to_nmodl(last_expression_statement));
103  ++it;
104  }
105  }
106  return it;
107 }

109 /**
110  * Check if provided statemenet is local variable declaration statement
111  * @param statement AST node representing statement in the MOD file
112  * @return True if statement is local variable declaration else False
113  *
114  * Statement declaration could be wrapped into another statement type like
115  * expression statement and hence we try to look inside if it's really a
116  * variable declaration.
117  */
118 static bool is_local_statement(std::shared_ptr<ast::Statement> statement) {
119  if (statement->is_local_list_statement()) {
120  return true;
121  }
122  if (statement->is_expression_statement()) {
123  auto e_statement = std::dynamic_pointer_cast<ast::ExpressionStatement>(statement);
124  auto expression = e_statement->get_expression();
125  if (expression->is_local_list_statement()) {
126  return true;
127  }
128  }
129  return false;
130 }

132 std::string& SympySolverVisitor::replaceAll(std::string& context,
133  const std::string& from,
134  const std::string& to) const {
135  std::size_t lookHere = 0;
136  std::size_t foundHere;
137  while ((foundHere = context.find(from, lookHere)) != std::string::npos) {

139  lookHere = foundHere + to.size();

156 std::string SympySolverVisitor::suffix_random_string(const std::string& original_string) const {
157  std::string new_string = original_string;
158  std::string random_string;
159  auto singleton_random_string_class = nmodl::utils::SingletonRandomString<4>::instance();
160  // Check if there is a variable defined in the mod file as original_string and if yes
161  // try to use a different string for the matrices created by sympy in the form
162  // <original_string>_<random_string>
163  while (vars.find(new_string) != vars.end()) {
164  random_string = singleton_random_string_class->reset_random_string(original_string);
165  new_string = original_string;
166  new_string += "_" + random_string;
167  }
168  return new_string;
169 }

171 void SympySolverVisitor::construct_eigen_solver_block(
172  const std::vector<std::string>& pre_solve_statements,
173  const std::vector<std::string>& solutions,
174  bool linear) {
175  // Provide random string to append to X, J, Jm and F matrices that
176  // are produced by sympy
177  std::string unique_X = suffix_random_string("X");
178  std::string unique_J = suffix_random_string("J");
179  std::string unique_Jm = suffix_random_string("Jm");
180  std::string unique_F = suffix_random_string("F");
181 
182  // filter solutions for matrices named "X", "J", "Jm" and "F" and change them to
183  // unique_X, unique_J, unique_Jm and unique_F respectively
184  auto solutions_filtered = filter_string_vector(solutions, "X[", unique_X + "[");
185  solutions_filtered = filter_string_vector(solutions_filtered, "J[", unique_J + "[");
186  solutions_filtered = filter_string_vector(solutions_filtered, "Jm[", unique_Jm + "[");
187  solutions_filtered = filter_string_vector(solutions_filtered, "F[", unique_F + "[");
188 
189  // find out where to insert solution in statement block
190  auto& statements = block_with_expression_statements->statements;
191  auto it = get_solution_location_iterator(statements);
192  // insert pre-solve statements below last linear eq in block
193  for (const auto& statement: pre_solve_statements) {
194  logger->debug("SympySolverVisitor :: -> adding statement: {}", statement);
195  it = statements.insert(it, create_statement(statement));
196  ++it;
197  }
198  // make Eigen vector <-> state var assignments
199  std::vector<std::string> setup_x_eqs;
200  std::vector<std::string> update_state_eqs;
201  for (int i = 0; i < state_vars.size(); i++) {
202  auto statement = state_vars[i] + " = " + unique_X + "[" + std::to_string(i) + "]";
203  auto rev_statement = unique_X + "[" + std::to_string(i) + "] = " + state_vars[i];
204  update_state_eqs.push_back(statement);
205  setup_x_eqs.push_back(rev_statement);
206  logger->debug("SympySolverVisitor :: setup_", unique_X, ": {}", rev_statement);
207  logger->debug("SympySolverVisitor :: update_state: {}", statement);
208  }
209 
210  for (const auto& sol: solutions_filtered) {
211  logger->debug("SympySolverVisitor :: -> adding statement: {}", sol);
212  }
213  // statements after last diff/linear/non-linear eq statement go into finalize_block
214  ast::StatementVector finalize_statements{it, statements.end()};
215  // remove them from the statement block
216  statements.erase(it, statements.end());
217  // also remove diff/linear/non-linear eq statements from the statement block
218  remove_statements_from_block(block_with_expression_statements, expression_statements);
219  // move any local variable declarations into variable_block
220  ast::StatementVector variable_statements;
221  // remaining statements in block should go into initialize_block
222  ast::StatementVector initialize_statements;
223  for (auto s: statements) {
224  if (is_local_statement(s)) {
225  variable_statements.push_back(s);
226  } else {
227  initialize_statements.push_back(s);
228  }
229  }
230  // make statement blocks
231  auto n_state_vars = std::make_shared<ast::Integer>(state_vars.size(), nullptr);
232  auto variable_block = std::make_shared<ast::StatementBlock>(std::move(variable_statements));
233  auto initialize_block = std::make_shared<ast::StatementBlock>(std::move(initialize_statements));
234  auto update_state_block = create_statement_block(update_state_eqs);
235  auto finalize_block = std::make_shared<ast::StatementBlock>(std::move(finalize_statements));
236 
237  if (linear) {
238  /// create eigen linear solver block
239  setup_x_eqs.insert(setup_x_eqs.end(), solutions_filtered.begin(), solutions_filtered.end());
240  auto setup_x_block = create_statement_block(setup_x_eqs);
241  auto solver_block = std::make_shared<ast::EigenLinearSolverBlock>(n_state_vars,
242  variable_block,
243  initialize_block,
244  setup_x_block,
245  update_state_block,
246  finalize_block);
247  /// replace statement block with solver block as it contains all statements
248  ast::StatementVector solver_block_statements{
249  std::make_shared<ast::ExpressionStatement>(solver_block)};
250  block_with_expression_statements->set_statements(std::move(solver_block_statements));
251  } else {
252  /// create eigen newton solver block
253  auto setup_x_block = create_statement_block(setup_x_eqs);
254  auto functor_block = create_statement_block(solutions_filtered);
255  auto solver_block = std::make_shared<ast::EigenNewtonSolverBlock>(n_state_vars,
256  variable_block,
257  initialize_block,
258  setup_x_block,
259  functor_block,
260  update_state_block,
261  finalize_block);
262  /// replace statement block with solver block as it contains all statements
263  ast::StatementVector solver_block_statements{
264  std::make_shared<ast::ExpressionStatement>(solver_block)};
265  block_with_expression_statements->set_statements(std::move(solver_block_statements));
266  }
267 }

269 void SympySolverVisitor::solve_linear_system(const std::vector<std::string>& pre_solve_statements) {
270  // construct ordered vector of state vars used in linear system
271  init_state_vars_vector();
272  // call sympy linear solver
273  bool small_system = (eq_system.size() <= SMALL_LINEAR_SYSTEM_MAX_STATES);
274  auto locals = py::dict("eq_strings"_a = eq_system,
275  "state_vars"_a = state_vars,
276  "vars"_a = vars,
277  "small_system"_a = small_system,
278  "do_cse"_a = elimination,
279  "function_calls"_a = function_calls);
280  py::exec(R"(
281  from nmodl.ode import solve_lin_system
282  exception_message = ""
283  try:
284  solutions, new_local_vars = solve_lin_system(eq_strings,
285  state_vars,
286  vars,
287  function_calls,
288  small_system,
289  do_cse)
290  except Exception as e:
291  # if we fail, fail silently and return empty string
292  solutions = [""]
293  new_local_vars = [""]
294  exception_message = str(e)
295  )",
296  py::globals(),
297  locals);
298  // returns a vector of solutions, i.e. new statements to add to block:
299  auto solutions = locals["solutions"].cast<std::vector<std::string>>();
300  // and a vector of new local variables that need to be declared in the block:
301  auto new_local_vars = locals["new_local_vars"].cast<std::vector<std::string>>();
302  // may also return a python exception message:
303  auto exception_message = locals["exception_message"].cast<std::string>();
304  if (!exception_message.empty()) {
305  logger->warn("SympySolverVisitor :: solve_lin_system python exception: " +
306  exception_message);
307  return;
308  }
309  // find out where to insert solutions in statement block
310  auto& statements = block_with_expression_statements->statements;
311  auto it = get_solution_location_iterator(statements);
312  if (small_system) {
313  // for small number of state vars, linear solver
314  // directly returns solution by solving symbolically at compile time
315  logger->debug("SympySolverVisitor :: Solving *small* linear system of eqs");
316  // declare new local vars
317  if (!new_local_vars.empty()) {
318  for (const auto& new_local_var: new_local_vars) {
319  logger->debug("SympySolverVisitor :: -> declaring new local variable: {}",
320  new_local_var);
321  add_local_variable(block_with_expression_statements, new_local_var);
322  }
323  }
324  // insert pre-solve statements below last linear eq in block
325  for (const auto& statement: pre_solve_statements) {
326  logger->debug("SympySolverVisitor :: -> adding statement: {}", statement);
327  it = statements.insert(it, create_statement(statement));
328  ++it;
329  }
330  // then insert new solution statements
331  for (const auto& sol: solutions) {
332  logger->debug("SympySolverVisitor :: -> adding statement: {}", sol);
333  it = statements.insert(it, create_statement(sol));
334  ++it;
335  }
336  /// remove original lineq statements from the block
337  remove_statements_from_block(block_with_expression_statements, expression_statements);
338  } else {
339  // otherwise it returns a linear matrix system to solve
340  logger->debug("SympySolverVisitor :: Constructing linear newton solve block");
341  construct_eigen_solver_block(pre_solve_statements, solutions, true);
342  }
343 }

403 void SympySolverVisitor::visit_diff_eq_expression(ast::DiffEqExpression* node) {
404  auto& lhs = node->get_expression()->lhs;
405  auto& rhs = node->get_expression()->rhs;
406 
407  if (!lhs->is_var_name()) {
408  logger->warn("SympySolverVisitor :: LHS of differential equation is not a VariableName");
409  return;
410  }
411  auto lhs_name = std::dynamic_pointer_cast<ast::VarName>(lhs)->get_name();
412  if ((lhs_name->is_indexed_name() &&
413  !std::dynamic_pointer_cast<ast::IndexedName>(lhs_name)->get_name()->is_prime_name()) ||
414  (!lhs_name->is_indexed_name() && !lhs_name->is_prime_name())) {
415  logger->warn("SympySolverVisitor :: LHS of differential equation is not a PrimeName");
416  return;
417  }
418 
419  check_expr_statements_in_same_block();
420 
421  const auto node_as_nmodl = to_nmodl_for_sympy(node);
422  const auto locals = py::dict("equation_string"_a = node_as_nmodl,
423  "dt_var"_a = codegen::naming::NTHREAD_DT_VARIABLE,
424  "vars"_a = vars,
425  "use_pade_approx"_a = use_pade_approx,
426  "function_calls"_a = function_calls);
427 
428  if (solve_method == codegen::naming::EULER_METHOD) {
429  logger->debug("SympySolverVisitor :: EULER - solving: {}", node_as_nmodl);
430  // replace x' = f(x) differential equation
431  // with forwards Euler timestep:
432  // x = x + f(x) * dt
433  py::exec(R"(
434  from nmodl.ode import forwards_euler2c
435  exception_message = ""
436  try:
437  solution = forwards_euler2c(equation_string, dt_var, vars, function_calls)
438  except Exception as e:
439  # if we fail, fail silently and return empty string
440  solution = ""
441  exception_message = str(e)
442  )",
443  py::globals(),
444  locals);
445  } else if (solve_method == codegen::naming::CNEXP_METHOD) {
446  // replace x' = f(x) differential equation
447  // with analytic solution for x(t+dt) in terms of x(t)
448  // x = ...
449  logger->debug("SympySolverVisitor :: CNEXP - solving: {}", node_as_nmodl);
450  py::exec(R"(
451  from nmodl.ode import integrate2c
452  exception_message = ""
453  try:
454  solution = integrate2c(equation_string, dt_var, vars,
455  use_pade_approx)
456  except Exception as e:
457  # if we fail, fail silently and return empty string
458  solution = ""
459  exception_message = str(e)
460  )",
461  py::globals(),
462  locals);
463  } else {
464  // for other solver methods: just collect the ODEs & return
465  std::string eq_str = to_nmodl_for_sympy(node);
466  std::string var_name = lhs_name->get_node_name();
467  if (lhs_name->is_indexed_name()) {
468  auto index_name = std::dynamic_pointer_cast<ast::IndexedName>(lhs_name);
469  var_name +=
470  "[" +
471  std::to_string(
472  std::dynamic_pointer_cast<ast::Integer>(index_name->get_length())->eval()) +
473  "]";
474  }
475  logger->debug("SympySolverVisitor :: adding ODE system: {}", eq_str);
476  eq_system.push_back(eq_str);
477  logger->debug("SympySolverVisitor :: adding state var: {}", var_name);
478  state_vars_in_block.insert(var_name);
479  expression_statements.insert(current_expression_statement);
480  last_expression_statement = current_expression_statement;
481  return;
482  }
483 
484  // replace ODE with solution in AST
485  auto solution = locals["solution"].cast<std::string>();
486  logger->debug("SympySolverVisitor :: -> solution: {}", solution);
487 
488  auto exception_message = locals["exception_message"].cast<std::string>();
489  if (!exception_message.empty()) {
490  logger->warn("SympySolverVisitor :: python exception: " + exception_message);
491  return;
492  }
493 
494  if (!solution.empty()) {
495  replace_diffeq_expression(node, solution);
496  } else {
497  logger->warn("SympySolverVisitor :: solution to differential equation not possible");
498  }
499 }

525 void SympySolverVisitor::visit_derivative_block(ast::DerivativeBlock* node) {
526  /// clear information from previous block, get global vars + block local vars
527  init_block_data(node);
528 
529  // get user specified solve method for this block
530  solve_method = derivative_block_solve_method[node->get_node_name()];
531 
532  // visit each differential equation:
533  // - for CNEXP or EULER, each equation is independent & is replaced with its solution
534  // - otherwise, each equation is added to eq_system
535  node->visit_children(*this);
536 
537  if (eq_system_is_valid && !eq_system.empty()) {
538  // solve system of ODEs in eq_system
539  logger->debug("SympySolverVisitor :: Solving {} system of ODEs", solve_method);
540 
541  // construct implicit Euler equations from ODEs
542  std::vector<std::string> pre_solve_statements;
543  for (auto& eq: eq_system) {
544  auto split_eq = stringutils::split_string(eq, '=');
545  auto x_prime_split = stringutils::split_string(split_eq[0], '\'');
546  auto x = stringutils::trim(x_prime_split[0]);
547  std::string x_array_index = "";
548  std::string x_array_index_i = "";
549  if (x_prime_split.size() > 1 && stringutils::trim(x_prime_split[1]).size() > 2) {
550  x_array_index = stringutils::trim(x_prime_split[1]);
551  x_array_index_i = "_" + x_array_index.substr(1, x_array_index.size() - 2);
552  }
553  std::string state_var_name = x + x_array_index;
554  auto var_eq_pair = conserve_equation.find(state_var_name);
555  if (var_eq_pair != conserve_equation.cend()) {
556  // replace the ODE for this state var with corresponding CONSERVE equation
557  eq = state_var_name + " = " + var_eq_pair->second;
558  logger->debug(

560  state_var_name,
561  var_eq_pair->second);
562  } else {
563  // no CONSERVE equation, construct Euler equation
564  auto dxdt = stringutils::trim(split_eq[1]);
565  auto old_x = "old_" + x + x_array_index_i; // TODO: do this properly,
566  // check name is unique
567  // declare old_x
568  logger->debug("SympySolverVisitor :: -> declaring new local variable: {}", old_x);
569  add_local_variable(block_with_expression_statements, old_x);
570  // assign old_x = x
571  pre_solve_statements.push_back(old_x + " = " + x + x_array_index);
572  // replace ODE with Euler equation
573  eq = x + x_array_index + " = " + old_x + " + " +
574  codegen::naming::NTHREAD_DT_VARIABLE + " * (" + dxdt + ")";
575  logger->debug("SympySolverVisitor :: -> constructed Euler eq: {}", eq);
576  }
577  }
578 
579  if (solve_method == codegen::naming::SPARSE_METHOD) {
580  solve_linear_system(pre_solve_statements);
581  } else if (solve_method == codegen::naming::DERIVIMPLICIT_METHOD) {
582  solve_non_linear_system(pre_solve_statements);
583  } else {
584  logger->error("SympySolverVisitor :: Solve method {} not supported", solve_method);
585  }
586  }

659 void SympySolverVisitor::visit_program(ast::Program* node) {
660  derivative_block_solve_method.clear();
661 
662  global_vars = get_global_vars(node);
663 
664  // get list of solve statements with names & methods
665  AstLookupVisitor ast_lookup_visitor;
666  auto solve_block_nodes = ast_lookup_visitor.lookup(node, ast::AstNodeType::SOLVE_BLOCK);
667  for (const auto& block: solve_block_nodes) {
668  if (auto block_ptr = std::dynamic_pointer_cast<ast::SolveBlock>(block)) {
669  const auto block_name = block_ptr->get_block_name()->get_value()->eval();
670  if (block_ptr->get_method()) {
671  // Note: solve method name is an optional parameter
672  // LINEAR and NONLINEAR blocks do not have solve method specified
673  const auto& solve_method = block_ptr->get_method()->get_value()->eval();
674  logger->debug("SympySolverVisitor :: Found SOLVE statement: using {} for {}",
675  solve_method,
676  block_name);
677  derivative_block_solve_method[block_name] = solve_method;
678  }
679  }
680  }
681 
682  // get set of all state vars
683  all_state_vars.clear();
684  if (auto symtab = node->get_symbol_table()) {
685  auto statevars = symtab->get_variables_with_properties(NmodlType::state_var);
686  for (const auto& v: statevars) {
687  std::string var_name = v->get_name();
688  if (v->is_array()) {
689  for (int i = 0; i < v->get_length(); ++i) {
690  std::string var_name_i = var_name + "[" + std::to_string(i) + "]";
691  all_state_vars.push_back(var_name_i);
692  }
693  } else {
694  all_state_vars.push_back(var_name);
695  }
696  }
697  }
698 
699  node->visit_children(*this);
700 }
