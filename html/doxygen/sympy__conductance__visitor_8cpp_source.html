28 /**
29  * Analyse breakpoint block to check if it is safe to insert CONDUCTANCE statements
30  *
31  * Most of the mod files have simple breakpoint blocks without any control flow
32  * statements. SympyConductanceVisitor just collects all the statements in the
33  * breakpoint block and insert conductance statements. If there are control flow
34  * statements like `IF { a } ELSE { b }` block with conflicting current statements
35  * inside IF and ELSE blocks or VERBATIM block then the resulting CONDUCTANCE
36  * statements may be incorrect. For now the simple approach is to not generate
37  * CONDUCTANCE statements if if-else statements exist in the block.
38  *
39  * @param node Ast node for breakpoint block
40  * @return true if it is safe to insert conductance statements otherwise false
41  */
42 static bool conductance_statement_possible(ast::BreakpointBlock* node) {
43  AstLookupVisitor v({AstNodeType::IF_STATEMENT, AstNodeType::VERBATIM});
44  node->accept(v);
45  return v.get_nodes().empty();
46 }

49 // Generate statement strings to be added to BREAKPOINT section
50 std::vector<std::string> SympyConductanceVisitor::generate_statement_strings(
51  ast::BreakpointBlock* node) {
52  std::vector<std::string> statements;
53 
54  // instead of passing all variables in the symbol table find out variables
55  // that are used in the current block and then pass to sympy
56  // name could be parameter or unit so check if it exist in symbol table
57  auto names_in_block = AstLookupVisitor().lookup(node, AstNodeType::NAME);
58  string_set used_names_in_block;
59  for (auto& name: names_in_block) {
60  auto varname = name->get_node_name();
61  if (all_vars.find(varname) != all_vars.end()) {
62  used_names_in_block.insert(varname);
63  }
64  }
65 
66  // iterate over binary expression lhs's from breakpoint
67  for (const auto& lhs_str: ordered_binary_exprs_lhs) {
68  // look for a current name that matches lhs of expr (current write name)
69  auto it = i_name.find(lhs_str);
70  if (it != i_name.end()) {
71  const auto& equation_string = ordered_binary_exprs[binary_expr_index[lhs_str]];
72  std::string i_name_str = it->second;
73  // SymPy needs the current expression & all previous expressions
74  std::vector<std::string> expressions(ordered_binary_exprs.begin(),
75  ordered_binary_exprs.begin() +
76  binary_expr_index[lhs_str] + 1);
77  // differentiate dI/dV
78  auto locals = py::dict("expressions"_a = expressions, "vars"_a = used_names_in_block);
79  py::exec(R"(
80  from nmodl.ode import differentiate2c
81  exception_message = ""
82  try:
83  rhs = expressions[-1].split("=", 1)[1]
84  solution = differentiate2c(rhs,
85  "v",
86  vars,
87  expressions[:-1]
88  )
89  except Exception as e:
90  # if we fail, fail silently and return empty string
91  solution = ""
92  exception_message = str(e)
93  )",
94  py::globals(),
95  locals);
96  auto dIdV = locals["solution"].cast<std::string>();
97  auto exception_message = locals["exception_message"].cast<std::string>();
98  if (!exception_message.empty()) {
99  logger->warn("SympyConductance :: python exception: {}", exception_message);
100  }
101  if (dIdV.empty()) {
102  logger->warn(
103  "SympyConductance :: analytic differentiation of ionic current "
104  "not possible");
105  } else {
106  std::string g_var = dIdV;
107  // if conductance g_var is not an existing variable, need to generate
108  // a new variable name, declare it, and assign it
109  if (all_vars.find(g_var) == all_vars.end()) {
110  // generate new variable name
111  std::map<std::string, int> var_map;
112  for (auto const& v: all_vars) {
113  var_map[v] = 0;
114  }
115  g_var = get_new_name("g", i_name_str, var_map);
116  // declare it
117  add_local_variable(node->get_statement_block().get(), g_var);
118  // asign dIdV to it
119  std::string statement_str = g_var;
120  statement_str.append(" = ").append(dIdV);
121  statements.insert(statements.begin(), statement_str);
122  logger->debug("SympyConductance :: Adding BREAKPOINT statement: {}",
123  statement_str);
124  }
125  std::string statement_str = "CONDUCTANCE " + g_var;
126  if (!i_name_str.empty()) {
127  statement_str += " USEION " + i_name_str;
128  }
129  statements.push_back(statement_str);
130  logger->debug("SympyConductance :: Adding BREAKPOINT statement: {}", statement_str);
131  }
132  }
133  }
134  return statements;
135 }

137 void SympyConductanceVisitor::visit_binary_expression(ast::BinaryExpression* node) {
138  // only want binary expressions from breakpoint block
139  if (!under_breakpoint_block) {
140  return;
141  }
142  // only want binary expressions of form x = ...
143  if (node->lhs->is_var_name() && (node->op.get_value() == BinaryOp::BOP_ASSIGN)) {
144  auto lhs_str =
145  std::dynamic_pointer_cast<ast::VarName>(node->lhs)->get_name()->get_node_name();
146  binary_expr_index[lhs_str] = ordered_binary_exprs.size();
147  ordered_binary_exprs.push_back(to_nmodl_for_sympy(node));
148  ordered_binary_exprs_lhs.push_back(lhs_str);
149  }
150 }

152 void SympyConductanceVisitor::lookup_nonspecific_statements() {
153  // add NONSPECIFIC_CURRENT statements to i_name map between write vars and names
154  // note that they don't have an ion name, so we set it to ""
155  if (!NONSPECIFIC_CONDUCTANCE_ALREADY_EXISTS) {
156  for (const auto& ns_curr_ast: nonspecific_nodes) {
157  logger->debug("SympyConductance :: Found NONSPECIFIC_CURRENT statement");
158  for (const auto& write_name:
159  std::dynamic_pointer_cast<ast::Nonspecific>(ns_curr_ast).get()->get_currents()) {
160  std::string curr_write = write_name->get_node_name();
161  logger->debug("SympyConductance :: -> Adding non-specific current write name: {}",
162  curr_write);
163  i_name[curr_write] = "";
164  }
165  }
166  }

169 void SympyConductanceVisitor::lookup_useion_statements() {
170  // add USEION statements to i_name map between write vars and names
171  for (const auto& useion_ast: use_ion_nodes) {
172  auto ion = std::dynamic_pointer_cast<ast::Useion>(useion_ast).get();
173  std::string ion_name = ion->get_node_name();
174  logger->debug("SympyConductance :: Found USEION statement {}", to_nmodl_for_sympy(ion));
175  if (i_ignore.find(ion_name) != i_ignore.end()) {
176  logger->debug("SympyConductance :: -> Ignoring ion current name: {}", ion_name);
177  } else {
178  for (const auto& w: ion->get_writelist()) {
179  std::string ion_write = w->get_node_name();
180  logger->debug(
181  "SympyConductance :: -> Adding ion write name: {} for ion current name: {}",
182  ion_write,
183  ion_name);
184  i_name[ion_write] = ion_name;
185  }
186  }
187  }

190 void SympyConductanceVisitor::visit_conductance_hint(ast::ConductanceHint* node) {
191  // find existing CONDUCTANCE statements - do not want to alter them
192  // so keep a set of ion names i_ignore that we should ignore later
193  logger->debug("SympyConductance :: Found existing CONDUCTANCE statement: {}",
194  to_nmodl_for_sympy(node));
195  if (auto ion = node->get_ion()) {
196  logger->debug("SympyConductance :: -> Ignoring ion current name: {}", ion->get_node_name());
197  i_ignore.insert(ion->get_node_name());
198  } else {
199  logger->debug("SympyConductance :: -> Ignoring all non-specific currents");
200  NONSPECIFIC_CONDUCTANCE_ALREADY_EXISTS = true;
201  }

204 void SympyConductanceVisitor::visit_breakpoint_block(ast::BreakpointBlock* node) {
205  // return if it's not safe to insert conductance statements
206  if (!conductance_statement_possible(node)) {
207  logger->warn("SympyConductance :: Unsafe to insert CONDUCTANCE statement");
208  return;
209  }
210 
211  // add any breakpoint local variables to vars
212  if (auto* symtab = node->get_statement_block()->get_symbol_table()) {
213  for (const auto& localvar: symtab->get_variables_with_properties(NmodlType::local_var)) {
214  all_vars.insert(localvar->get_name());
215  }
216  }
217  // visit BREAKPOINT block statements
218  under_breakpoint_block = true;
219  node->visit_children(*this);
220  under_breakpoint_block = false;
221 
222  // lookup USEION and NONSPECIFIC statements from NEURON block
223  lookup_useion_statements();
224  lookup_nonspecific_statements();
225 
226  // add new CONDUCTANCE statements to BREAKPOINT
227  auto new_statements = generate_statement_strings(node);
228  if (!new_statements.empty()) {
229  // get a copy of existing BREAKPOINT statements
230  auto brkpnt_statements = node->get_statement_block()->get_statements();
231  // insert new CONDUCTANCE statements at top of BREAKPOINT
232  // or just below LOCAL statement if it exists
233  auto insertion_point = brkpnt_statements.begin();
234  while ((*insertion_point)->is_local_list_statement()) {
235  ++insertion_point;
236  }
237  for (const auto& statement_str: new_statements) {
238  insertion_point = brkpnt_statements.insert(insertion_point,
239  create_statement(statement_str));
240  }
241  // replace old set of BREAKPOINT statements in AST with new one
242  node->get_statement_block()->set_statements(std::move(brkpnt_statements));
243  }
244 }
