114  /// Takes as parameter an std::shared_ptr<ast::Ast> node
115  /// and returns a relative error with the name, the type
116  /// and the location of the unhandled statement
117  ///
118  /// \tparam T Type of node parameter in the ast::Ast
119  /// \param node Not used by the function
120  /// \param ast_node Ast node which is checked
121  /// \return std::string error
122  template <typename T>
123  std::string return_error_with_name(ast::Ast* node, std::shared_ptr<ast::Ast>& ast_node) {
124  auto real_type_block = std::dynamic_pointer_cast<T>(ast_node);
125  return "\"{}\" {}construct found at [{}] is not handled\n"_format(
126  ast_node->get_node_name(),
127  real_type_block->get_nmodl_name(),
128  real_type_block->get_token()->position());
129  }

131  /// Takes as parameter an std::shared_ptr<ast::Ast> node
