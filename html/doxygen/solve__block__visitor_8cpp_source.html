27 /**
28  * Create solution expression node that will be used for solve block
29  * @param solve_block solve block used to describe node to solve and method
30  * @return solution expression that will be used to replace the solve block
31  *
32  * Depending on the solver used, solve block is converted to solve expression statement
33  * that will be used to replace solve block. Note that the blocks are clones instead of
34  * shared_ptr because DerivimplicitCallback is currently contain whole node
35  * instead of just pointer.
36  */
37 ast::SolutionExpression* SolveBlockVisitor::create_solution_expression(
38  ast::SolveBlock* solve_block) {
39  /// find out the block that is going to solved
40  std::string block_name = solve_block->get_block_name()->get_node_name();
41  auto solve_node_symbol = symtab->lookup(block_name);
42  assert(solve_node_symbol != nullptr);
43  auto node_to_solve = solve_node_symbol->get_node();
44 
45  /// in case of derivimplicit method if neuron solver is used (i.e. not sympy) then
46  /// the solution is not in place but we have to create a callback to newton solver
47  auto method = solve_block->get_method();
48  std::string solve_method = method ? method->get_node_name() : "";
49  if (solve_method == codegen::naming::DERIVIMPLICIT_METHOD &&
50  !has_sympy_solution(node_to_solve)) {
51  /// typically derivimplicit is used for derivative block only
52  assert(node_to_solve->get_node_type() == ast::AstNodeType::DERIVATIVE_BLOCK);
53  auto derivative_block = dynamic_cast<ast::DerivativeBlock*>(node_to_solve);
54  auto callback_expr = new ast::DerivimplicitCallback(derivative_block->clone());
55  return new ast::SolutionExpression(solve_block->clone(), callback_expr);
56  }
57 
58  auto block_to_solve = node_to_solve->get_statement_block();
59  return new ast::SolutionExpression(solve_block->clone(), block_to_solve->clone());
60 }

62 /**
63  * Replace solve blocks with solution expression
64  * @param node Ast node for SOLVE statement in the mod file
65  */
66 void SolveBlockVisitor::visit_expression_statement(ast::ExpressionStatement* node) {
67  node->visit_children(*this);
68  if (node->get_expression()->is_solve_block()) {
69  auto solve_block = dynamic_cast<ast::SolveBlock*>(node->get_expression().get());
70  auto sol_expr = create_solution_expression(solve_block);
71  if (in_breakpoint_block) {
72  nrn_state_solve_statements.emplace_back(new ast::ExpressionStatement(sol_expr));
73  } else {
74  node->set_expression(std::shared_ptr<ast::SolutionExpression>(sol_expr));
75  }
76  }
77 }
