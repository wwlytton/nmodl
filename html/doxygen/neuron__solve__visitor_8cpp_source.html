45 void NeuronSolveVisitor::visit_binary_expression(ast::BinaryExpression* node) {
46  auto& lhs = node->lhs;
47  auto& rhs = node->rhs;
48  auto& op = node->op;
49 
50  /// we have to only solve odes under derivative block where lhs is variable
51  if (!derivative_block || !differential_equation || !lhs->is_var_name()) {
52  return;
53  }
54 
55  auto solve_method = solve_blocks[derivative_block_name];
56  auto name = std::dynamic_pointer_cast<ast::VarName>(lhs)->get_name();
57 
58  if (name->is_prime_name()) {
59  auto equation = to_nmodl(node);
60  parser::DiffeqDriver diffeq_driver;
61 
62  if (solve_method == codegen::naming::CNEXP_METHOD) {
63  std::string solution;
64  /// check if ode can be solved with cnexp method
65  if (diffeq_driver.cnexp_possible(equation, solution)) {
66  auto statement = create_statement(solution);
67  auto expr_statement = std::dynamic_pointer_cast<ast::ExpressionStatement>(
68  statement);
69  auto bin_expr = std::dynamic_pointer_cast<ast::BinaryExpression>(
70  expr_statement->get_expression());
71  lhs.reset(bin_expr->lhs->clone());
72  rhs.reset(bin_expr->rhs->clone());
73  } else {
74  logger->warn("NeuronSolveVisitor :: cnexp solver not possible for {}",
75  to_nmodl(node));
76  }
77  } else if (solve_method == codegen::naming::EULER_METHOD) {
78  std::string solution = diffeq_driver.solve(equation, solve_method);
79  auto statement = create_statement(solution);
80  auto expr_statement = std::dynamic_pointer_cast<ast::ExpressionStatement>(statement);
81  auto bin_expr = std::dynamic_pointer_cast<ast::BinaryExpression>(
82  expr_statement->get_expression());
83  lhs.reset(bin_expr->lhs->clone());
84  rhs.reset(bin_expr->rhs->clone());
85  } else if (solve_method == codegen::naming::DERIVIMPLICIT_METHOD) {
86  auto varname = "D" + name->get_node_name();
87  auto variable = new ast::Name(new ast::String(varname));
88  lhs.reset(variable);
89  if (program_symtab->lookup(varname) == nullptr) {
90  auto symbol = std::make_shared<symtab::Symbol>(varname, ModToken());
91  symbol->set_original_name(name->get_node_name());
92  symbol->created_from_state();
93  program_symtab->insert(symbol);
94  }
95  } else {
96  logger->error("NeuronSolveVisitor :: solver method '{}' not supported", solve_method);
97  }
98  }
99 }
