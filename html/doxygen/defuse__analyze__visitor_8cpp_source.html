83 /** Evaluate sub-blocks like if, elseif and else
84  * As these are innermost blocks, we have to just check first use
85  * of variable in this block and that's the result of this block.
86  */
87 DUState DUInstance::sub_block_eval() {
88  DUState result = DUState::NONE;
89  for (auto& chain: children) {
90  auto child_state = chain.eval();
91  if (child_state == DUState::U || child_state == DUState::D) {
92  result = child_state;
93  break;
94  }
95  if (child_state == DUState::CD) {
96  result = DUState::CD;
97  }
98  }
99  return result;
100 }

102 /**
103  * Evaluate conditional block containing sub-blocks like if, elseif and else
104  *
105  * Note that sub-blocks are already evaluated by sub_block_eval() and has only
106  * leaf nodes. In order to find effective defuse, following rules are used:
107  *
108  * - If variable is "used" in any of the sub-block then it's effectively
109  * "U". This is because any branch can be taken at runtime.
110  *
111  * - If variable is "defined" in all sub-blocks doesn't mean that it's
112  * effectively "D". This is because if we can have just "if-elseif"
113  * which could never be taken. Same for empty "if". In order to decide
114  * if it is "D", we make sure there is no empty block and there must
115  * be "else" block with "D". Note that "U" definitions are already
116  * covered in 1) and hence this rule is safe.
117  *
118  * - If there is an "if" with "D" or empty "if" followed by "D" in "else"
119  * block, we can't say it's definition. In this case we return "CD".
120  *
121  * - If there is empty "if" followed by "U" in "else" block, we can say
122  * it's "use". This is because we don't want to "localize" such variables.
123  *
124  * - If we reach else block with either D or CD and if there is no empty
125  * block encountered, this means every block has either "D" or "CD". In
126  * this case we can say that entire block effectively has "D".
127  */
128 DUState DUInstance::conditional_block_eval() {
129  DUState result = DUState::NONE;
130  bool block_with_none = false;
131  bool block_non_def_cdef = false;
132 
133  for (auto& chain: children) {
134  auto child_state = chain.eval();
135  if (child_state == DUState::U) {
136  result = child_state;
137  break;
138  }
139  if (child_state == DUState::NONE) {
140  block_with_none = true;
141  }
142  if (child_state == DUState::D || child_state == DUState::CD) {
143  result = DUState::CD;
144  if (chain.state == DUState::ELSE && !block_with_none) {
145  result = DUState::D;
146  break;
147  }
148  }
149  }
150  return result;
151 }

153 /** Find "effective" usage of variable from def-use chain.
154  * Note that we are interested in "global" variable usage
155  * and hence we consider only [U,D] states and not [LU, LD]
156  */
157 DUState DUInstance::eval() {
158  auto result = state;
159  if (state == DUState::IF || state == DUState::ELSEIF || state == DUState::ELSE) {
160  result = sub_block_eval();
161  } else if (state == DUState::CONDITIONAL_BLOCK) {
162  result = conditional_block_eval();
163  }
164  return result;
165 }

167 /// first usage of a variable in a block decides whether it's definition
168 /// or usage. Note that if-else blocks already evaluated.
169 DUState DUChain::eval() {
170  auto result = DUState::NONE;
171  for (auto& inst: chain) {
172  auto re = inst.eval();
173  if (re == DUState::U || re == DUState::D) {
174  result = re;
175  break;
176  }
177  if (re == DUState::CD) {
178  result = re;
179  }
180  }
181  return result;
182 }

229 void DefUseAnalyzeVisitor::visit_if_statement(ast::IfStatement* node) {
230  /// store previous chain
231  auto previous_chain = current_chain;
232 
233  /// starting new if block
234  previous_chain->push_back(DUInstance(DUState::CONDITIONAL_BLOCK));
235  current_chain = &(previous_chain->back().children);
236 
237  /// visiting if sub-block
238  auto last_chain = current_chain;
239  start_new_chain(DUState::IF);
240  node->get_condition()->accept(*this);
241  auto block = node->get_statement_block();
242  if (block) {
243  block->accept(*this);
244  }
245  current_chain = last_chain;
246 
247  /// visiting else if sub-blocks
248  for (const auto& item: node->get_elseifs()) {
249  visit_with_new_chain(item.get(), DUState::ELSEIF);
250  }
251 
252  /// visiting else sub-block
253  if (node->get_elses()) {
254  visit_with_new_chain(node->get_elses().get(), DUState::ELSE);
255  }
256 
257  /// restore to previous chain
258  current_chain = previous_chain;
259 }

273 /**
274  * Update the Def-Use chain for given variable
275  *
276  * @param name Name of the variable excluding index or dimension
277  *
278  * Update def-use chain if we encounter a variable that we are looking for.
279  * If we encounter non-supported construct then we mark that variable as "use"
280  * because we haven't completely analyzed the usage. Marking that variable "U"
281  * make sures that won't get optimized. Then we distinguish between local and
282  * non-local variables. All variables that appear on lhs are marked as "definitions"
283  * whereas the one on rhs are marked as "usages".
284  */
285 void DefUseAnalyzeVisitor::update_defuse_chain(const std::string& name) {
286  auto symbol = current_symtab->lookup_in_scope(name);
287  assert(symbol != nullptr);
288  // variable properties that make variable local
289  auto properties = NmodlType::local_var | NmodlType::argument;
290  auto is_local = symbol->has_any_property(properties);
291 
292  if (unsupported_node) {
293  current_chain->push_back(DUInstance(DUState::U));
294  } else if (visiting_lhs) {
295  if (is_local) {
296  current_chain->push_back(DUInstance(DUState::LD));
297  } else {
298  current_chain->push_back(DUInstance(DUState::D));
299  }
300  } else {
301  if (is_local) {
302  current_chain->push_back(DUInstance(DUState::LU));
303  } else {
304  current_chain->push_back(DUInstance(DUState::U));
305  }
306  }
307 }
