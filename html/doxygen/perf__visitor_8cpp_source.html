25 /// count math operations from all binary expressions
26 void PerfVisitor::visit_binary_expression(ast::BinaryExpression* node) {
27  bool assign_op = false;
28 
29  if (start_measurement) {
30  auto value = node->get_op().get_value();
31  switch (value) {
32  case ast::BOP_ADDITION:
33  current_block_perf.n_add++;
34  break;
35 
36  case ast::BOP_SUBTRACTION:
37  current_block_perf.n_sub++;
38  break;
39 
40  case ast::BOP_MULTIPLICATION:
41  current_block_perf.n_mul++;
42  break;
43 
44  case ast::BOP_DIVISION:
45  current_block_perf.n_div++;
46  break;
47 
48  case ast::BOP_POWER:
49  current_block_perf.n_pow++;
50  break;
51 
52  case ast::BOP_AND:
53  current_block_perf.n_and++;
54  break;
55 
56  case ast::BOP_OR:
57  current_block_perf.n_or++;
58  break;
59 
60  case ast::BOP_GREATER:
61  current_block_perf.n_gt++;
62  break;
63 
64  case ast::BOP_GREATER_EQUAL:
65  current_block_perf.n_ge++;
66  break;
67 
68  case ast::BOP_LESS:
69  current_block_perf.n_lt++;
70  break;
71 
72  case ast::BOP_LESS_EQUAL:
73  current_block_perf.n_le++;
74  break;
75 
76  case ast::BOP_ASSIGN:
77  current_block_perf.n_assign++;
78  assign_op = true;
79  break;
80 
81  case ast::BOP_NOT_EQUAL:
82  current_block_perf.n_ne++;
83  break;
84 
85  case ast::BOP_EXACT_EQUAL:
86  current_block_perf.n_ee++;
87  break;
88 
89  default:
90  throw std::logic_error("Binary operator not handled in perf visitor");
91  }
92  }
93 
94  /// if visiting assignment expression, symbols from lhs
95  /// are written and hence need flag to track
96  if (assign_op) {
97  visiting_lhs_expression = true;
98  }
99 
100  node->get_lhs()->accept(*this);
101 
102  /// lhs is done (rhs is read only)
103  visiting_lhs_expression = false;
104 
105  node->get_rhs()->accept(*this);
106 }

119 /** Helper function used by all ast nodes : visit all children
120  * recursively and performance stats get added on stack. Once
121  * all children visited, we get total performance by summing
122  * perfstat of all children.
123  */
124 void PerfVisitor::measure_performance(ast::Ast* node) {
125  start_measurement = true;
126 
127  node->visit_children(*this);
128 
129  PerfStat perf;
130  while (!children_blocks_perf.empty()) {
131  perf = perf + children_blocks_perf.top();
132  children_blocks_perf.pop();
133  }
134 
135  auto symtab = node->get_symbol_table();
136  if (symtab == nullptr) {
137  throw std::runtime_error("Perfvisitor : symbol table not setup for " +
138  node->get_node_type_name());
139  }
140 
141  auto name = symtab->name();
142  if (node->is_derivative_block()) {
143  name = node->get_node_type_name();
144  }
145 
146  if (printer) {
147  printer->push_block(name);
148  }
149 
150  perf.title = "Performance Statistics of " + name;
151  perf.print(stream);
152 
153  if (printer) {
154  add_perf_to_printer(perf);
155  printer->pop_block();
156  }
157 
158  start_measurement = false;
159 
160  /// clear var usage map
161  for (auto& var_set: var_usage) {
162  var_set.second.clear();
163  }
164 }

166 /// count function calls and "most useful" or "commonly used" math functions
167 void PerfVisitor::visit_function_call(ast::FunctionCall* node) {
168  under_function_call = true;
169 
170  if (start_measurement) {
171  auto name = node->get_node_name();
172  if (name == "exp") {
173  current_block_perf.n_exp++;
174  } else if (name == "log") {
175  current_block_perf.n_log++;
176  } else if (name == "pow") {
177  current_block_perf.n_pow++;
178  }
179  node->visit_children(*this);
180 
181  auto symbol = current_symtab->lookup_in_scope(name);
182  auto method_property = NmodlType::procedure_block | NmodlType::function_block;
183  if (symbol != nullptr && symbol->has_any_property(method_property)) {
184  current_block_perf.n_int_func_call++;
185  } else {
186  current_block_perf.n_ext_func_call++;
187  }
188  }
189 
190  under_function_call = false;
191 }

219 void PerfVisitor::count_variables() {
220  /// number of instance variables: range or assigned variables
221  /// one caveat is that the global variables appearing in
222  /// assigned block are not treated as range
223  num_instance_variables = 0;
224 
225  NmodlType property = NmodlType::range_var | NmodlType::assigned_definition |
226  NmodlType::state_var;
227  auto variables = current_symtab->get_variables_with_properties(property);
228 
229  for (auto& variable: variables) {
230  if (!variable->has_any_property(NmodlType::global_var)) {
231  num_instance_variables++;
232  if (variable->has_any_property(NmodlType::param_assign)) {
233  num_constant_instance_variables++;
234  }
235  if (variable->has_any_status(Status::localized)) {
236  num_localized_instance_variables++;
237  }
238  }
239  }
240 
241  /// state variables have state_var property
242  property = NmodlType::state_var;
243  variables = current_symtab->get_variables_with_properties(property);
244  num_state_variables = variables.size();
245 
246  /// pointer variables have pointer/bbcorepointer
247  property = NmodlType::pointer_var | NmodlType::bbcore_pointer_var;
248  variables = current_symtab->get_variables_with_properties(property);
249  num_pointer_variables = variables.size();
250 
251 
252  /// number of global variables : parameters and pointers could appear also
253  /// as range variables and hence need to filter out. But if anything declared
254  /// as global is always global.
255  property = NmodlType::global_var | NmodlType::param_assign | NmodlType::bbcore_pointer_var |
256  NmodlType::pointer_var;
257  variables = current_symtab->get_variables_with_properties(property);
258  num_global_variables = 0;
259  for (auto& variable: variables) {
260  auto is_global = variable->has_any_property(NmodlType::global_var);
261  property = NmodlType::range_var | NmodlType::assigned_definition;
262  if (!variable->has_any_property(property) || is_global) {
263  num_global_variables++;
264  if (variable->has_any_property(NmodlType::param_assign)) {
265  num_constant_global_variables++;
266  }
267  if (variable->has_any_status(Status::localized)) {
268  num_localized_global_variables++;
269  }
270  }
271  }

338 /** Blocks like function can have multiple statement blocks and
339  * blocks like net receive has nested initial blocks. Hence need
340  * to maintain separate stack.
341  */
342 void PerfVisitor::visit_statement_block(ast::StatementBlock* node) {
343  /// starting new block, store current state
344  blocks_perf.push(current_block_perf);
345 
346  current_symtab = node->get_symbol_table();
347 
348  if (current_symtab == nullptr) {
349  throw std::runtime_error("Perfvisitor : symbol table not setup for " +
350  node->get_node_type_name());
351  }
352 
353  /// new block perf starts from zero
354  current_block_perf = PerfStat();
355 
356  node->visit_children(*this);
357 
358  /// add performance of all visited children
359  total_perf = total_perf + current_block_perf;
360 
361  children_blocks_perf.push(current_block_perf);
362 
363  // go back to parent block's state
364  current_block_perf = blocks_perf.top();
365  blocks_perf.pop();
366 }

397 /** Certain statements / symbols needs extra check while measuring
398  * read/write operations. For example, for expression "exp(a+b)",
399  * "exp" is an external math function and we should not increment read
400  * count for "exp" symbol. Same for solve statement where name will
401  * be derivative block name and neuron solver method.
402  */
403 bool PerfVisitor::symbol_to_skip(const std::shared_ptr<Symbol>& symbol) {
404  bool skip = false;
405 
406  auto is_method = symbol->has_any_property(NmodlType::extern_method | NmodlType::function_block);
407  if (is_method && under_function_call) {
408  skip = true;
409  }
410 
411  is_method = symbol->has_any_property(NmodlType::derivative_block | NmodlType::extern_method);
412  if (is_method && under_solve_block) {
413  skip = true;
414  }
415 
416  return skip;
417 }

439 /** Find symbol in closest scope (up to parent) and update
440  * read/write count. Also update ops count in current block.
441  */
442 void PerfVisitor::update_memory_ops(const std::string& name) {
443  if (!start_measurement || current_symtab == nullptr) {
444  return;
445  }
446 
447  auto symbol = current_symtab->lookup_in_scope(name);
448  if (symbol == nullptr || symbol_to_skip(symbol)) {
449  return;
450  }
451 
452  if (is_local_variable(symbol)) {
453  if (visiting_lhs_expression) {
454  symbol->write();
455  current_block_perf.n_local_write++;
456  } else {
457  symbol->read();
458  current_block_perf.n_local_read++;
459  }
460  return;
461  }
462 
463  /// lhs symbols get written
464  if (visiting_lhs_expression) {
465  symbol->write();
466  if (is_constant_variable(symbol)) {
467  current_block_perf.n_constant_write++;
468  if (var_usage[const_memw_key].find(name) == var_usage[const_memw_key].end()) {
469  current_block_perf.n_unique_constant_write++;
470  var_usage[const_memw_key].insert(name);
471  }
472  } else {
473  current_block_perf.n_global_write++;
474  if (var_usage[global_memw_key].find(name) == var_usage[global_memw_key].end()) {
475  current_block_perf.n_unique_global_write++;
476  var_usage[global_memw_key].insert(name);
477  }
478  }
479  return;
480  }
481 
482  /// rhs symbols get read
483  symbol->read();
484  if (is_constant_variable(symbol)) {
485  current_block_perf.n_constant_read++;
486  if (var_usage[const_memr_key].find(name) == var_usage[const_memr_key].end()) {
487  current_block_perf.n_unique_constant_read++;
488  var_usage[const_memr_key].insert(name);
489  }
490  } else {
491  current_block_perf.n_global_read++;
492  if (var_usage[global_memr_key].find(name) == var_usage[global_memr_key].end()) {
493  current_block_perf.n_unique_global_read++;
494  var_usage[global_memr_key].insert(name);
495  }
496  }
497 }
