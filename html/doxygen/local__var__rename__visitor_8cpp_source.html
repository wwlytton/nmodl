18 /// rename name conflicting variables in the statement block and it's all children
19 void LocalVarRenameVisitor::visit_statement_block(ast::StatementBlock* node) {
20  /// nothing to do
21  if (node->get_statements().empty()) {
22  return;
23  }
24 
25  auto current_symtab = node->get_symbol_table();
26  if (current_symtab != nullptr) {
27  symtab = current_symtab;
28  }
29 
30  // Some statements like forall, from, while are of type expression statement type.
31  // These statements contain statement block but do not have symbol table. And hence
32  // we push last non-null symbol table on the stack.
33  symtab_stack.push(symtab);
34 
35  // first need to process all children : perform recursively from innermost block
36  for (const auto& item: node->get_statements()) {
37  item->visit_children(*this);
38  }
39 
40  /// go back to previous block in hierarchy
41  symtab = symtab_stack.top();
42  symtab_stack.pop();
43 
44  SymbolTable* parent_symtab = nullptr;
45  if (symtab != nullptr) {
46  parent_symtab = symtab->get_parent_table();
47  }
48 
49  auto variables = get_local_variables(node);
50 
51  /// global blocks do not change (do no have parent symbol table)
52  /// if no variables in the block then there is nothing to do
53  if (parent_symtab == nullptr || variables == nullptr) {
54  return;
55  }
56 
57  RenameVisitor rename_visitor;
58 
59  for (const auto& var: *variables) {
60  std::string name = var->get_node_name();
61  auto s = parent_symtab->lookup_in_scope(name);
62  /// if symbol is a variable name (avoid renaming use of units like mV)
63  if (s && s->is_variable()) {
64  std::string new_name = get_new_name(name, "r", renamed_variables);
65  rename_visitor.set(name, new_name);
66  rename_visitor.visit_statement_block(node);
67  auto symbol = symtab->lookup_in_scope(name);
68  symbol->set_name(new_name);
69  symbol->mark_renamed();
70  }
71  }
72 }
