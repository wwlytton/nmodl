63 std::string CodegenCompatibilityVisitor::return_error_if_no_bbcore_read_write(
64  ast::Ast* node,
65  std::shared_ptr<ast::Ast>& ast_node) {
66  std::stringstream error_message_no_bbcore_read_write;
67  auto verbatim_nodes = AstLookupVisitor().lookup(node, AstNodeType::VERBATIM);
68  auto found_bbcore_read = false;
69  auto found_bbcore_write = false;
70  for (const auto& it: verbatim_nodes) {
71  auto verbatim = std::dynamic_pointer_cast<ast::Verbatim>(it);
72 
73  auto verbatim_statement_string = verbatim->get_statement()->get_value();
74 
75  // Parse verbatim block to find out if there is a token "bbcore_read" or
76  // "bbcore_write". If there is not, then the function is not defined or
77  // is commented.
78  parser::CDriver driver;
79 
80  driver.scan_string(verbatim_statement_string);
81  auto tokens = driver.all_tokens();
82 
83  for (const auto& token: tokens) {
84  if (token == "bbcore_read") {
85  found_bbcore_read = true;
86  }
87  if (token == "bbcore_write") {
88  found_bbcore_write = true;
89  }
90  }
91  }
92  if (!found_bbcore_read) {
93  error_message_no_bbcore_read_write
94  << "\"bbcore_read\" function not defined in any VERBATIM block\n";
95  }
96  if (!found_bbcore_write) {
97  error_message_no_bbcore_read_write
98  << "\"bbcore_write\" function not defined in any VERBATIM block\n";
99  }
100  return error_message_no_bbcore_read_write.str();
101 }

109 bool CodegenCompatibilityVisitor::find_unhandled_ast_nodes(Ast* node) {
110  std::vector<ast::AstNodeType> unhandled_ast_types;
111  for (auto kv: unhandled_ast_types_func) {
112  unhandled_ast_types.push_back(kv.first);
113  }
114  unhandled_ast_nodes = AstLookupVisitor().lookup(node, unhandled_ast_types);
115 
116  std::stringstream ss;
117  for (auto it: unhandled_ast_nodes) {
118  auto node_type = it->get_node_type();
119  ss << (this->*unhandled_ast_types_func[node_type])(node, it);
120  }
121  if (!ss.str().empty()) {
122  logger->error("Code incompatibility detected");
123  logger->error("Cannot translate mod file to .cpp file");
124  logger->error("Fix the following errors and try again");
125  std::string line;
126  std::istringstream ss_stringstream(ss.str());
127  while (std::getline(ss_stringstream, line)) {
128  if (!line.empty())
129  logger->error("Code Incompatibility :: {}"_format(line));
130  }
131  return true;
132  }
133  return false;
134 }
